496 char *parse_url(const char *param)
497 {
498  size_t url_len = strlen(param);
499  url = malloc(url_len+1);
500  assert(url != NULL);
501  strcpy(url, param);
502 
503  bool is_https = false;
504  bool is_http = false;
505 
506  for (int i = 0; i < url_len; i++) {
507  if (domain == NULL) {
508  if (i < strlen(HTTP) && tolower(url[i]) == HTTP[i]) {
509  // eat it
510  is_http = true;
511  } else if (i < strlen(HTTPS) && tolower(url[i]) == HTTPS[i]) {
512  //
513  is_https = true;
514  } else if (i == strlen(HTTP) && is_http) {
515  // is an HTTP url starting at url[i]
516  domain = &url[i];
517  } else if (i == strlen(HTTPS) && is_https) {
518  // is an HTTPS url starting at url[i]
519  domain = &url[i];
520  } else if (i < strlen(HTTP)) {
521  domain = url;
522  } else {
523  fprintf(stderr, "ERROR: Malformed domain '%s'\n", url);
524  exit(EXIT_FAILURE);
525  }
526  } else if (url[i] == ':') {
527  portstr = &url[i+1];
528  url[i] = '\0';
529  } else if (url[i] == '/') {
530  path = &url[i+1];
531  url[i] = '\0';
532  break;
533  }
534  }
535 
536  if (portstr == NULL || portstr == "") {
537  portstr = is_https ? "443" : "80";
538  }
539 
540  assert(domain != NULL);
541  if (path == NULL) {
542  path = "";
543  }
544 
545  if (path[0] == '/') {
546  // abc://sdf.com/ causes the '/sdf.com/' to become path and 'abc' to be domain
547  fprintf(stderr, "ERROR: Unknown protocol '%s'", url);
548  exit(EXIT_FAILURE);
549  }
550 
551  if (strchr(domain, '.') == NULL || strlen(domain) < 4) {
552  // Must at least be a.bc or 0.0.0.0
553  fprintf(stderr, "ERROR: Malformed domain '%s'\n", domain);
554  exit(EXIT_FAILURE);
555  }
556 
557  if (strlen(domain) >= sizeof(uxchAT_urlHostStr_t)) {
558  fprintf(stderr, "ERROR: Domain contains more than %d characters!\n", sizeof(uxchAT_urlHostStr_t)-1);
559  exit(EXIT_FAILURE);
560  }
561 
562  int res = sscanf(portstr, "%u", &port_number);
563  if (res != 1 || port_number < 1 || port_number > 65535) {
564  fprintf(stderr, "ERROR: Invalid portstr '%s'!\n", portstr);
565  exit(EXIT_FAILURE);
566  }
567 
568  printf("GET using protocol %s from domain '%s' port %u path '/%s'\n",
569  is_https ? "HTTPS" : "HTTP",
570  domain,
571  port_number,
572  path
573  );
574 
575  get_command = malloc(sizeof(GET_COMMAND) + strlen(path));
576  assert(get_command != NULL);
577  sprintf(get_command, GET_COMMAND, path);
578 
579  host_header = malloc(sizeof(HOST_HEADER) + strlen(domain));
580  assert(host_header != NULL);
581  sprintf(host_header, HOST_HEADER, domain);
582 
583  return url;
584 }

592  }
593

597  } else if (strcmp(argv[a], "--password") == 0 && a+1 < argc) {
598  ap_passphrase = argv[++a];
599  } else if (strcmp(argv[a], "--passphrase") == 0 && a+1 < argc) {
600  ap_passphrase = argv[++a];
601  } else if (strcmp(argv[a], "--timeout") == 0 && a+1 < argc) {
602  if (sscanf(argv[++a], "%d", &timeout_seconds) != 1) {
603  timeout_seconds = -1;
604  }
605  } else if (url == NULL) {
606  if (a < argc) {
607  url = parse_url(argv[a]);
608  assert(url != NULL);
609  }
610  } else {
611  print_usage();
612  exit(EXIT_FAILURE);
613  }
614  }
